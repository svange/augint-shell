32def _build_dep_sync_prefix() -> str: 

33 """Build shell commands to sync project dependencies before launching Claude. 

34 

35 Handles both Python (uv) and Node.js (npm) projects. Runs in the pane's 

36 working directory (set by tmux ``-c``), so it detects the project type from 

37 the files present there. 

38 

39 Output is intentionally terse -- ``tail -3`` keeps only the summary line(s). 

40 """ 

41 return ( 

42 "if [ -f pyproject.toml ]; then uv sync 2>&1 | tail -3; fi; " 

43 "if [ -f package-lock.json ]; then npm ci --loglevel=warn 2>&1 | tail -3; " 

44 "elif [ -f package.json ]; then npm install --loglevel=warn 2>&1 | tail -3; fi; " 

45 )

48def build_claude_pane_command( 

49 *, 

50 repo_name: str, 

51 safe: bool = False, 

52 extra_args: tuple[str, ...] = (), 

53 worktree_name: str | None = None, 

54 sync_deps: bool = True, 

55) -> str: 

56 """Build the claude invocation string for a single tmux pane. 

57 

58 Runs directly inside the container. Exports ``UV_PROJECT_ENVIRONMENT`` 

59 so each repo gets an isolated virtualenv within the shared UV cache volume. 

60 Uses Claude Code's ``-n`` flag for session naming. 

61 

62 When *worktree_name* is set, appends ``-wt-{worktree_name}`` to the venv 

63 path so each worktree gets its own isolated virtualenv. 

64 

65 When *sync_deps* is True (default), prepends ``uv sync`` / ``npm ci`` 

66 commands so the per-repo venv is initialised before Claude starts. 

67 

68 In non-safe mode, tries ``claude -c`` (continue previous conversation) 

69 first; falls back to a fresh session if that fails (e.g. no prior 

70 conversation exists). 

71 """ 

72 uv_env = f"UV_PROJECT_ENVIRONMENT={uv_venv_path(repo_name, worktree_name)}" 

73 

74 dep_prefix = _build_dep_sync_prefix() if sync_deps else "" 

75 

76 if safe: 

77 parts: list[str] = ["claude", "-n", repo_name] 

78 if extra_args: 

79 parts.append("--") 

80 parts.extend(extra_args) 

81 cmd = " ".join(shlex.quote(p) for p in parts) 

82 return f"export {uv_env}; {dep_prefix}{cmd}" 

83 

84 # Non-safe: build two commands -- continue attempt and fresh fallback 

85 base_parts: list[str] = ["claude", "--dangerously-skip-permissions", "-n", repo_name] 

86 continue_parts: list[str] = [ 

87 "claude", 

88 "--dangerously-skip-permissions", 

89 "-c", 

90 "-n", 

91 repo_name, 

92 ] 

93 

94 extra_suffix = "" 

95 if extra_args: 

96 extra_suffix = " -- " + " ".join(shlex.quote(a) for a in extra_args) 

97 

98 continue_cmd = " ".join(shlex.quote(p) for p in continue_parts) + extra_suffix 

99 fresh_cmd = " ".join(shlex.quote(p) for p in base_parts) + extra_suffix 

100 

101 return f"export {uv_env}; {dep_prefix}{continue_cmd} || {fresh_cmd}"

104def build_check_session_command(container_name: str, session_name: str) -> list[str]: 

105 """Build a command to check whether a tmux session exists in a container.

108 session exists; non-zero means it does not.

110 return ["docker", "exec", container_name, "tmux", "has-session", "-t", session_name]

131def select_layout(pane_count: int) -> str: 

132 """Return the tmux layout name for the given pane count. 

133 

134 2 panes: even-vertical (top / bottom) 

135 3 panes: main-horizontal (1 large top, 2 split bottom -- first pane ~65%) 

136 4 panes: tiled (even quarters) 

137 """ 

138 layouts = { 

139 2: "even-vertical", 

140 3: "main-horizontal", 

141 4: "tiled", 

142 } 

143 return layouts.get(pane_count, "tiled")

146def build_tmux_commands( 

147 container_name: str, 

148 session_name: str, 

149 panes: list[PaneSpec], 

150) -> list[list[str]]: 

151 """Build the sequence of ``docker exec`` commands for the tmux session. 

152 

153 Every command except the **last** is non-interactive (no ``-it``). 

154 The last command is an interactive ``docker exec -it ... tmux attach`` 

155 intended to be executed via ``os.execvp`` to replace the current process. 

156 

157 Returns a list of argument lists suitable for ``subprocess.run()``. 

158 """ 

159 if not panes: 

160 return [] 

161 

162 cmds: list[list[str]] = [] 

163 

164 def _exec(*args: str) -> list[str]: 

165 return ["docker", "exec", container_name, *args] 

166 

167 # 1. Kill stale session (ignore errors -- caller should use check=False) 

168 cmds.append(_exec("tmux", "kill-session", "-t", session_name)) 

169 

170 # 2. Create session with first pane 

171 first = panes[0] 

172 cmds.append(_exec("tmux", "new-session", "-d", "-s", session_name, "-c", first.working_dir)) 

173 

174 # 3. Split additional panes 

175 for pane in panes[1:]: 

176 cmds.append( 

177 _exec("tmux", "split-window", "-t", f"{session_name}:0", "-c", pane.working_dir) 

178 ) 

179 

180 # 4. Apply layout 

181 layout = select_layout(len(panes)) 

182 cmds.append(_exec("tmux", "select-layout", "-t", f"{session_name}:0", layout)) 

183 

184 # 4b. For 3-pane main-horizontal: give the first pane ~65% of height 

185 if len(panes) == 3: 

186 cmds.append(_exec("tmux", "set-option", "-t", session_name, "main-pane-height", "65%")) 

187 # Re-apply layout so the new main-pane-height takes effect 

188 cmds.append(_exec("tmux", "select-layout", "-t", f"{session_name}:0", layout)) 

189 

190 # 5. Set pane titles 

191 for i, pane in enumerate(panes): 

192 cmds.append(_exec("tmux", "select-pane", "-t", f"{session_name}:0.{i}", "-T", pane.name)) 

193 

194 # 6. Send commands to each pane 

195 for i, pane in enumerate(panes): 

196 cmds.append( 

197 _exec("tmux", "send-keys", "-t", f"{session_name}:0.{i}", pane.command, "Enter") 

198 ) 

199 

200 # 7. Configure session options -- warm amber/mauve scheme 

201 # 

202 # Colour palette (designed to complement Claude Code's warm UI): 

203 # colour172 (#d78700) = amber -> active pane, session name 

204 # colour95 (#875f5f) = mauve -> inactive panes, help text 

205 # colour235 (#262626) = dark bg -> status bar background 

206 # colour248 (#a8a8a8) = light -> status bar text 

207 # 

208 # Three distinct visual bands: amber=active, mauve=inactive, 

209 # gray=Claude Code UI. No pair blends. 

210 session_options: list[tuple[str, str]] = [ 

211 # Mouse & responsiveness 

212 ("mouse", "on"), 

213 ("escape-time", "10"), 

214 ("history-limit", "50000"), 

215 ("focus-events", "on"), 

216 # Pane borders: amber active, dusty mauve inactive, heavy lines 

217 ("pane-border-status", "top"), 

218 ("pane-border-lines", "heavy"), 

219 ( 

220 "pane-border-format", 

221 "#{?pane_active,#[fg=colour172 bold] #{pane_title} ,#[fg=colour95] #{pane_title} }", 

222 ), 

223 ("pane-border-style", "fg=colour95"), 

224 ("pane-active-border-style", "fg=colour172,bold"), 

225 ("pane-border-indicators", "arrows"), 

226 # Status bar 

227 ("status-style", "bg=colour235 fg=colour248"), 

228 ("status-left", "#[fg=colour172,bold] #S #[fg=colour248]| "), 

229 ("status-right", "#[fg=colour95] C-b z=zoom C-b d=detach "), 

230 ("status-left-length", "40"), 

231 ("status-right-length", "40"), 

232 ] 

233 for key, value in session_options: 

234 cmds.append(_exec("tmux", "set-option", "-t", session_name, key, value)) 

235 

236 # 8. Server-level terminal options for true-color support 

237 cmds.append(_exec("tmux", "set-option", "-s", "default-terminal", "tmux-256color")) 

238 cmds.append(_exec("tmux", "set-option", "-sa", "terminal-overrides", ",xterm*:Tc")) 

239 

240 # 9. Select first pane 

241 cmds.append(_exec("tmux", "select-pane", "-t", f"{session_name}:0.0")) 

242 

243 # 10. Final: interactive attach (caller should execvp this one) 

244 cmds.append( 

245 ["docker", "exec", "-it", container_name, "tmux", "attach-session", "-t", session_name] 

246 ) 

247 

248 return cmds
